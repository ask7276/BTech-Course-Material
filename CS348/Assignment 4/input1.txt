PROGRAM STATS
VAR
	SUM,SUMSQ,I,VALUE,MEAN,VARIANCE, MEAN: INTEGER ,
	S, T : REAL
BEGIN
	SUM := 0;
	SUMSQ := 0;
	T := 1.9;
	A := 1;
        S:= SUM;
	FOR I := 1 TO 100 DO
		BEGIN 
			READ(VALUE);
			SUM := SUM + VALUE;
			SUMSQ := SUMSQ + VALUE*VALUE
		END;
	MEAN := SUM DIV 100;
	VARIANCE := SUMSQ DIV 100 - MEAN * MEAN;
	WRITE(MEAN,VARIANCE,S)
END.
